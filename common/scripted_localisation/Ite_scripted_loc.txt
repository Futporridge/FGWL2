# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = GetITE_REZIST_CENTRAL
	text = {
		trigger = {
			check_variable = { ite_resistenza_central_italy > 9 }
		}
		localization_key = ITE_REZIST_CENTRAL_very_high
	}
	text = {
		trigger = {
			check_variable = { ite_resistenza_central_italy > 6 }
		}
		localization_key = ITE_REZIST_CENTRAL_high
	}
	text = {
		trigger = {
			check_variable = { ite_resistenza_central_italy > 3 }
		}
		localization_key = ITE_REZIST_CENTRAL_medium
	}
	text = {
		trigger = {
			check_variable = { ite_resistenza_central_italy > 1 }
		}
		localization_key = ITE_REZIST_CENTRAL_low
	}
	text = {
		localization_key = ITE_REZIST_CENTRAL_very_low
	}
}
